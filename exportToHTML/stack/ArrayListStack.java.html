1    package stack;
2    
3    import java.util.*;
4    /**
5     * Implementation of a stack using an ArrayList to hold the
6     * elements of the stack
7     * @author trao
8     *
9     * @param <T>
10    */
11   public class ArrayListStack<T> implements IStack<T>
12   {
13       //--------------------------------------------------
14       // Body of the stack is held in an ArrayList
15       private ArrayList<T> stackBody;
16   
17       //--------------------------------------------------
18       /**
19        * Constructor: Instantiates the body
20        */
21       public ArrayListStack()
22       {
23           stackBody = new ArrayList<T>();
24       }
25       //--------------------------------------------------
26       /**
27        * Test for emptiness: size == 0 means stack is empty
28        */
29       public boolean isEmpty()
30       {
31           return (stackBody.size() == 0);
32       }
33       //--------------------------------------------------
34       /**
35        * Insert a new item at the top of the stack
36        */
37       public void push(T item)
38       {
39           stackBody.add(item);
40       }
41       //--------------------------------------------------
42       /**
43        * Delete the top-most element of the stack
44        * Report error and return null if stack is empty
45        */
46       public T pop()
47       {
48           if (isEmpty())
49           {
50               System.out.println
51                       ("Error in ArrayStack.pop() Stack Empty ");
52               return null;
53           }
54           else
55           {
56               T topElement = stackBody.get(stackBody.size()-1);
57               stackBody.remove(stackBody.size()-1);
58               return topElement;
59           }
60       }
61       //--------------------------------------------------
62       /**
63        * Fetch the top-most element of the stack without
64        * removing it. Report error and return null if stack
65        * is empty
66        */
67       public T top()
68       {
69           if (isEmpty())
70           {
71               System.out.println
72                       ("Error in ArrayStack.top() Stack Empty ");
73               return null;
74           }
75           else
76           {
77               T topElement = stackBody.get(stackBody.size()-1);
78               return topElement;
79           }
80       }
81   
82   }
