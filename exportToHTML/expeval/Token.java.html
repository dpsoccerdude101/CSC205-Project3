4    /**
5     * Represents a token in an expression. Just provides some
6     * static methods
7     *
8     * @author (T.M. Rao)
9     */
10   public class Token {
11       //-------------------------------------------------------------
12       //It classifies the following as "OPERATORS". If you need more,
13       //you have to change this
14       private final static String[] validOperators =
15               {"+", "-", "*", "/", "%", "<", "<=", ">", ">=", "==", "!=", "&&", "||"};
16   
17       //-------------------------------------------------------------
18       private String body;
19   
20       //-------------------------------------------------------------
21       private final static char OPENPAREN = '(';
22       private final static char CLOSEDPAREN = ')';
23   
24   
25       //-------------------------------------------------------------
26       public Token(String tok) {
27           body = tok;
28       }
29   
30       //-------------------------------------------------------------
31       public String getBody() {
32           return body;
33       }
34       //-------------------------------------------------------------
35   
36       /**
37        * Takes a String argument and checks if the first
38        * char (i.e. charAt(0)) is an operator. The string
39        * of valid operators is defined above.
40        */
41       public boolean isOperator() {
42           for (int i = 0; i < validOperators.length; i++)
43               if (validOperators[i].equals(body))
44                   return true;
45           return false;
46       }
47   
48       //-------------------------------------------------------------
49   
50       /**
51        * Is the first char of the arg an open Parenthesis?
52        */
53       public boolean isOpenParen() {
54           char ch = body.charAt(0);
55           return (ch == OPENPAREN);
56       }
57   
58       //-------------------------------------------------------------
59   
60       /**
61        * Is the first char of the arg an close Parenthesis?
62        */
63       public boolean isClosedParen() {
64           char ch = body.charAt(0);
65           return (ch == CLOSEDPAREN);
66       }
67   
68       //-------------------------------------------------------------
69       public boolean isOperand() {
70           return (!((isOperator()
71                   || isOpenParen()
72                   || isClosedParen()
73           )));
74       }
75   
76       //-------------------------------------------------------------
77   
78       /**
79        * Defines the precedence values for some operators.
80        */
81       public int getPrecedence() {
82           if (body.equals("||")) return 3;
83           else if (body.equals("&&")) return 4;
84           else if (body.equals("==") || body.equals("!=")) return 8;
85           else if (body.equals("<") || body.equals("<=") || body.equals(">") || body.equals(">=")) return 9;
86           else if (body.equals("+") || body.equals("-")) return 11;
87           else if (body.equals("*") || body.equals("/") || body.equals("%")) return 12;
88           return -1;
89       }
90   
91       //--------------------------------------------------------------
92       public String toString() {
93           return body;
94       }
95   
96   }
