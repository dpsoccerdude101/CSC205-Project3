1    package expeval;
2    
3    /**
4     * This defines the concept of an Expression. The expression is
5     * implemented as an array of tokens: where each token contains
6     * a string.
7     *
8     * @author (T.M. Rao)
9     *
10    */
11   import java.util.*;
12   public class Expression
13   {
14       //----------------------------------------------------------------
15       private ArrayList<Token> expression;
16   //Expression is input as a String
17   
18   
19   
20       //----------------------------------------------------------------
21       /**
22        * Constructor: The String exp contains the infix expression.
23        * This breaks it into tokens and
24        * stores them in an ArrayList.
25        */
26       public Expression(String exp)
27       {
28           //Create the array that represents the body of the Expression
29           expression = new ArrayList<Token>();
30   
31   
32           //Tokenize the expression using the String Tokenizer class.
33           //Assumed that the tokens are separated by spaces
34           StringTokenizer strTok = new StringTokenizer(exp," ");
35   
36           //Scan the input expression from left to right for tokens
37           while (strTok.hasMoreElements())
38           {
39               Token tok = new Token((String) strTok.nextElement());
40               expression.add(tok);
41           }
42       }
43   
44       //-----------------------------------------------------------------
45       /**
46        * Alternative constructor. This creates an array with no tokens
47        * in it.
48        */
49       public Expression()
50       {
51           //Create the array that represents the body of the Expression
52           expression = new ArrayList<Token>();
53       }
54   
55       //----------------------------------------------------------------
56       public int size()
57       {
58           return expression.size();
59       }
60   
61   
62       //-----------------------------------------------------------------
63       /**
64        * Appending a token at the end of the expression
65        */
66       public void add(Token newElement)
67       {
68           expression.add(newElement);
69       }
70   
71       //-----------------------------------------------------------------
72       /**
73        * Creates a printable string from the expression
74        */
75       public String toString()
76       {
77           String ret = "";
78           for (int i = 0; i < expression.size(); i++)
79               ret = ret + expression.get(i).getBody()+" ";
80           return ret;
81       }
82   
83       //----------------------------------------------------------------
84       public ArrayList<Token> getExpression()
85       {
86           return expression;
87       }
88   }
