16           InfixToPostfixConverter itop =

22   
23           //Print the postfix
24           System.out.println
25                   ("Postfix Expression:     "+postfix.toString());
26   
27           //Instantiate a PostfixEvaluator
28           PostfixEvaluator peval = new PostfixEvaluator(postfix);
29   
30           //Evaluate the postfix expression
31           int value = peval.eval();

33           //Return the calculated value
34           return value;
